123 /*In color conversations:
124  * - When converting to bigger color type the LSB weight of 1 LSB is calculated
125  * E.g. 16 bit Red has 5 bits
126  * 8 bit Red has 2 bits
127  * ----------------------
128  * 8 bit red LSB = (2^5 - 1) / (2^2 - 1) = 31 / 3 = 10
129  *
130  * - When calculating to smaller color type simply shift out the LSBs
131  * E.g. 8 bit Red has 2 bits
132  * 16 bit Red has 5 bits
133  * ----------------------
134  * Shift right with 5 - 3 = 2
135  */

161 static inline uint8_t lv_color_to8(lv_color_t color) {
162 #if LV_COLOR_DEPTH == 1
163  if (color.full == 0)
164  return 0;
165  else
166  return 0xFF;
167 #elif LV_COLOR_DEPTH == 8
168  return color.full;
169 #elif LV_COLOR_DEPTH == 16
170  lv_color8_t ret;
171  ret.red = color.red >> 2; /* 5 - 3 = 2*/
172  ret.green = color.green >> 3; /* 6 - 3 = 3*/
173  ret.blue = color.blue >> 3; /* 5 - 2 = 3*/
174  return ret.full;
175 #elif LV_COLOR_DEPTH == 24
176  lv_color8_t ret;
177  ret.red = color.red >> 5; /* 8 - 3 = 5*/
178  ret.green = color.green >> 5; /* 8 - 3 = 5*/
179  ret.blue = color.blue >> 6; /* 8 - 2 = 6*/
180  return ret.full;
181 #endif
182 }
